package main

import (
	"fmt"
	"sync"
)

func main() {
	var (
		slc []int
		n   = 10
		wg  sync.WaitGroup
	)

	mutex := sync.Mutex{}

	wg.Add(n)
	for i := 0; i < n; i++ {
		go func() {
			mutex.Lock()
			slc = append(slc, i)
			mutex.Unlock()
			wg.Done()
		}()
	}
	wg.Wait()

	fmt.Println("len:", len(slc))
	fmt.Println("done")
}

package main

import (
	"fmt"
	"sync"
)
func main(){
	var (
		slc []int
		n   = 100
		wg  sync.WaitGroup
	)

	ch := make(chan int)
	doneCh := make(chan struct{})
	go func() {
		for v := range ch {
			slc = append(slc, v)
		}
		doneCh <- struct{}{}
	}()

	wg.Add(n)
	for i := 0; i < n; i++ {
		tmp := i
		go func() {
			ch <- tmp
			wg.Done()
		}()
	}

	wg.Wait()
	close(ch)
	<-doneCh

	fmt.Println("len:", len(slc))
	fmt.Println("done")

}
